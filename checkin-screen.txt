'use client'

import { useState } from 'react'
import { Mail, LogIn, Zap } from 'lucide-react'

interface CheckinScreenProps {
  onCheckIn: (customerId: string) => void
}

export default function CheckinScreen({ onCheckIn }: CheckinScreenProps) {
  const [email, setEmail] = useState('')
  const [loading, setLoading] = useState(false)

  const generateCustomerId = () => {
    return `RV${Date.now().toString().slice(-8)}`
  }

  const handleCheckIn = (e: React.FormEvent) => {
    e.preventDefault()
    if (!email.trim()) return

    setLoading(true)
    setTimeout(() => {
      const customerId = generateCustomerId()
      onCheckIn(customerId)
      setLoading(false)
    }, 800)
  }

  return (
    <div className="min-h-screen flex items-center justify-center px-4">
      <div className="w-full max-w-md animate-slide-in">
        {/* Header */}
        <div className="text-center mb-8">
          <div className="flex items-center justify-center gap-2 mb-4">
            <Zap className="w-8 h-8 text-indigo-600" />
            <h1 className="text-4xl font-bold bg-gradient-to-r from-indigo-600 to-pink-600 bg-clip-text text-transparent">
              RetailVision
            </h1>
          </div>
          <p className="text-gray-600 text-lg">AI-Powered Checkout System</p>
        </div>

        {/* Card */}
        <div className="card bg-white/80 backdrop-blur border-2 border-indigo-100 shadow-xl">
          <form onSubmit={handleCheckIn} className="space-y-6">
            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-3">
                Enter Your Email
              </label>
              <div className="relative">
                <Mail className="absolute left-3 top-3 w-5 h-5 text-indigo-400" />
                <input
                  type="email"
                  value={email}
                  onChange={(e) => setEmail(e.target.value)}
                  placeholder="you@example.com"
                  className="input-field pl-10 bg-white"
                  required
                />
              </div>
              <p className="text-xs text-gray-500 mt-2">
                Get your unique shopping ID for this session
              </p>
            </div>

            <button
              type="submit"
              disabled={loading || !email}
              className="w-full btn-primary flex items-center justify-center gap-2 disabled:opacity-50 disabled:cursor-not-allowed"
            >
              <LogIn className="w-5 h-5" />
              {loading ? 'Checking In...' : 'Check In to Store'}
            </button>
          </form>

          <div className="mt-6 pt-6 border-t border-gray-200">
            <p className="text-xs text-gray-600 text-center">
              ✓ Secure check-in process<br/>
              ✓ Real-time product detection<br/>
              ✓ Instant checkout experience
            </p>
          </div>
        </div>

        {/* Footer */}
        <div className="mt-8 text-center text-sm text-gray-600">
          <p>Skip long queues • Shop smarter • Pay faster</p>
        </div>
      </div>
    </div>
  )
}
